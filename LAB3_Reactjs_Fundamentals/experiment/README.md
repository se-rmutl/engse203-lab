  const [isDragging, setIsDragging] = useState(false);
  const [clickCount, setClickCount] = useState(0);
  const cardRef = useRef(null);

  const handleMouseDown = (e) => {
    setIsDragging(true);
    const rect = cardRef.current.getBoundingClientRect();
    setPosition({
      x: e.clientX - rect.left,
      y: e.clientY - rect.top
    });
  };

    if (!isDragging) return;

    setPosition({
      x: e.clientX - position.x,
      y: e.clientY - position.y
    });
  };

  const handleMouseUp = () => {
    setIsDragging(false);
  };

  const handleClick = () => {
    setClickCount(prev => prev + 1);
  };

  const handleDoubleClick = () => {
    setPosition({ x: 0, y: 0 });
    setClickCount(0);
  };

  const handleKeyDown = (e) => {
    const step = 10;

      case 'ArrowUp':
        setPosition(prev => ({ ...prev, y: prev.y - step }));

      case 'ArrowDown':
        setPosition(prev => ({ ...prev, y: prev.y + step }));

      case 'ArrowLeft':
        setPosition(prev => ({ ...prev, x: prev.x - step }));
        break;
      case 'ArrowRight':
        setPosition(prev => ({ ...prev, x: prev.x + step }));
        break;
      case 'Enter':
        handleClick();
        break;
      default:
        break;
    }
  };

## üîç Troubleshooting Common Issues

### 1. useState Updates ‡πÑ‡∏°‡πà‡∏ó‡∏≥‡∏á‡∏≤‡∏ô
```jsx
// ‚ùå Wrong
const [count, setCount] = useState(0);
setCount(count + 1);
setCount(count + 1); // Still 1, not 2

// ‚úÖ Correct
setCount(prev => prev + 1);
setCount(prev => prev + 1); // Now it's 2
```

### 2. useEffect Infinite Loop
```jsx
// ‚ùå Wrong - infinite loop
useEffect(() => {
  setData(newData);
}); // No dependency array

// ‚úÖ Correct
useEffect(() => {
  setData(newData);
}, []); // Empty dependency array
```

### 3. Context Value Updates
```jsx
// ‚ùå Wrong - creates new object every render
<MyContext.Provider value={{ user, setUser }}>

// ‚úÖ Correct - memoize the value
const value = useMemo(() => ({ user, setUser }), [user]);
<MyContext.Provider value={value}>
```
