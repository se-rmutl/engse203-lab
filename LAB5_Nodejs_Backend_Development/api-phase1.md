### üîÑ Status Update Route (30 ‡∏ô‡∏≤‡∏ó‡∏µ)

```javascript

const VALID_STATUSES = ['Available', 'Busy', 'Break', 'Offline'];

// Status transition rules
const VALID_TRANSITIONS = {
  'Available': ['Busy', 'Break', 'Offline'],
  'Busy': ['Available', 'Break'],
  'Break': ['Available', 'Offline'],
  'Offline': ['Available']
};

// PATCH - ‡∏≠‡∏±‡∏û‡πÄ‡∏î‡∏ó Agent Status
app.patch('/api/agents/:id/status', (req, res) => {
  const id = parseInt(req.params.id);
  const { status } = req.body;
  const agentIndex = agents.findIndex(a => a.id === id);
  
  // ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö Agent ‡∏°‡∏µ‡∏≠‡∏¢‡∏π‡πà‡πÑ‡∏´‡∏°
  if (agentIndex === -1) {
    return res.status(404).json({
      success: false,
      message: 'Agent not found'
    });
  }
  
  // ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö Status ‡∏ñ‡∏π‡∏Å‡∏ï‡πâ‡∏≠‡∏á‡πÑ‡∏´‡∏°
  if (!VALID_STATUSES.includes(status)) {
    return res.status(400).json({
      success: false,
      message: `Invalid status. Valid: ${VALID_STATUSES.join(', ')}`
    });
  }
  
  const agent = agents[agentIndex];
  const currentStatus = agent.status;
  
  // ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏Å‡∏≤‡∏£‡πÄ‡∏õ‡∏•‡∏µ‡πà‡∏¢‡∏ô‡∏™‡∏ñ‡∏≤‡∏ô‡∏∞‡πÑ‡∏î‡πâ‡πÑ‡∏´‡∏°
  if (!VALID_TRANSITIONS[currentStatus].includes(status)) {
    return res.status(400).json({
      success: false,
      message: `Cannot change from ${currentStatus} to ${status}`
    });
  }
  
  // ‡∏≠‡∏±‡∏û‡πÄ‡∏î‡∏ó‡∏™‡∏ñ‡∏≤‡∏ô‡∏∞
  agent.status = status;
  
  console.log(`Agent ${agent.agentCode}: ${currentStatus} ‚Üí ${status}`);
  
  res.json({
    success: true,
    message: `Status updated to ${status}`,
    data: agent
  });
});
```
